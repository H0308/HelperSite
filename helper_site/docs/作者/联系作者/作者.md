<script>
let items = document.querySelectorAll('.content-card');
items.forEach(item => {
    let isHovering = false;
    let resetTimer = null;
    const avatar = item.querySelector('.card-avatar img');
    
    item.addEventListener('mouseenter', () => {
        isHovering = true;
        clearTimeout(resetTimer);
        item.style.transition = 'transform 0.2s ease-out';
    });
    
    item.addEventListener('mouseleave', () => {
        isHovering = false;
        resetTimer = setTimeout(() => {
            item.style.transition = 'transform 0.6s cubic-bezier(0.34, 1.56, 0.64, 1)';
            item.style.setProperty('--rX', '0deg');
            item.style.setProperty('--rY', '0deg');
        }, 400);
    });

    item.addEventListener('mousemove', (e) => {
        if (!isHovering) return;
        
        clearTimeout(resetTimer);
        
        const rect = item.getBoundingClientRect();
        const x = (e.clientX - rect.left - rect.width/2) / (rect.width/2);
        const y = (e.clientY - rect.top - rect.height/2) / (rect.height/2);
        
        // 加大倾斜角度到12度
        const rotateY = Math.max(-12, Math.min(12, x * 12));
        const rotateX = Math.max(-12, Math.min(12, y * -12));
        
        item.style.transition = 'transform 0.1s linear';
        item.style.setProperty('--rX', rotateX + 'deg');
        item.style.setProperty('--rY', rotateY + 'deg');
    });
});
</script>
